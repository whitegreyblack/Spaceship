class Map:
	class Tile:
		var char. color
	var array = [[0->x]->y]
	
	def create_tile_map(self):
		for 0->y:
			for 0->x:
				array[y][x] = Tile(char, color)
	def output(self):
		for 0->y:
			for 0->x:
				return array[y][x]

class LocalMap(Map):
	# single child node making map into a 1d bi directional list
	def __init__(self, parent):
		var above/paren/prev = parent
		var below/child/next
	
	@property
	def next(self):
		return self.next
	@property
	def prev(self):
		return self.next
	@property.setter
	def next(self, l):
		self.next = l
	@property.setter
	def prev(self, l):
		self.prev = l
